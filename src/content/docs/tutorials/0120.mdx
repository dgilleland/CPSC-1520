    /**
     * Generates the string of Fibonacci numbers up to a
     * certain quantity.
     * @param {number} quantity The quantity of numbers in the sequence
     * @returns {string} - A string with the Fibonacci sequence
    */
    const buildFibonacciSequence = function(quantity) {
      let message;
      if(quantity <= 0) {
        message = 'Invalid quantity - must be greater than zero';
      } else if(quantity === 1) {
        message = '1';
      } else {
        // Calculate the series
        let previous = 0;
        let current = 1;
        message = '1'; // First in sequence
        for(let count = 2; count <= quantity; count++) {
          let next = current + previous;
          message += `, ${next}`;
          previous = current;
          current = next;
        }
      }

      return message;
    }
